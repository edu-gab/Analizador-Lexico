#!/usr/bin/env kotlin

fun main() {
    // Declaración de variables y operadores aritméticos
    var x = 10.8
    val y = 20.8f
    var result = 0

    result = x + y
    println(result)

    result = x - y
    println(result)

    result = x * y
    println(result)

    result = y / x
    println(result)

    // Bucle for
    for (i in 1..5) {
        println(i)
    }
}
